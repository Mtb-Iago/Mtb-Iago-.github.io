    constructor(valor, proximo){
        this.valor = valor;
        this.proximo = proximo;
    }
}

const listaSimplesmenteEncadeada = () => {
    
    const addInicio = (valor) => {
        let novoNo = new ListaClass(valor, this.cabeca);
        
        if(this.cabeca == null){
            this.cabeca = novoNo;
            this.calda = novoNo;
        } else{
            this.cabeca = novoNo;

    const addFim = (valor) => {
        let novoNo = new ListaClass(valor, null);

        if(this.cabeca == null){
            this.cabeca = novoNo;
            this.calda = novoNo;
        } else {
            this.calda.proximo = novoNo;
            this.calda = novoNo;
        }
    }

        if(this.cabeca == null){
            return;
        }

        if(this.cabeca.valor == valor){
            
            this.cabeca = this.cabeca.proximo;

        } else {

            let noAtual = this.cabeca;
            let noPorDeletar = null;

            while(noAtual.proximo != null){
                if(noAtual.proximo.valor == valor){
                    noPorDeletar = noAtual.proximo;
                    noAtual.proximo = noAtual.proximo.proximo;
                    break;
                }
                noAtual = noAtual.proximo;
            }

            if(noPorDeletar == null){
                console.log("Valor não encontrado");
            }

        }

    }

    const ler = () => {
        if(this.cabeca == null){
            console.log("Vazio");
            return;
        }

        //console.log("Listando");
        while(noAtual != null){

            console.log(noAtual.valor);

            noAtual = noAtual.proximo;
        }
    }
    return {
        addInicio,
        addFim,
        remove,
        ler
    }

}
